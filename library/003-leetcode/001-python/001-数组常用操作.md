# 数组常用操作

## 创建数组
```py
my_list = []

my_list2 = [1, 2, 3]

my_list3 = my_list + my_list2
```

## 添加元素
```py
my_list = [1, 2, 3]

my_list.append(4)
# 直接在末尾加，o(1)
# my_list = [1, 2, 3, 4]

my_list.insert(1, 9)
# 在指定索引元素的后面插入一个元素，o(n)
# my_list = [1, 2, 9, 3, 4]
```

## 修改元素
```py
my_list = [1, 2, 3]

my_list[1] = 4
# 修改指定索引的元素，o(1)
# my_list = [1, 4, 3]
```

## 获取元素
```py
my_list = [1, 2, 3]

print(my_list[1])
# 找到指定索引的元素，o(1)
```

## 删除元素
```py
my_list = [11, 12, 13, 14]

my_list.remove(12)
# 找到指定值的元素并删除，o(n)
# my_list = [11, 13, 14]

my_list.pop(1)
# 删除指定索引的元素，o(n)（删掉后需要挪后面的元素）
# my_list = [11, 14]

my_list.pop()
# 删除最后一个元素，o(1)
# my_list = [11]
```

## 获取数组长度
```py
my_list = [1, 2, 3]

list_length = len(my_list)
```

## 遍历数组
```py
my_list = [11, 12, 13]

# 遍历元素本身，11 12 13
for num in my_list:
  println(num)
"""
11
12
13
"""

# 遍历元素和元素
for index, num in enumerate(my_list):
  println(f'index: {index}, num: {num}')
"""
index: 1, num: 11
index: 2, num: 12
index: 3, num: 13
"""

# 根据索引遍历元素
for i in range(len(my_list)):
  println(f'index: {i}, num: {my_list[i]}')
"""
index: 1, num: 11
index: 2, num: 12
index: 3, num: 13
"""
```

## 查找某个元素
```py
my_list = [11, 12, 13]

index = my_list.index(13)
# 查找值等于指定值的元素的索引，o(1)
# index=2
```

## 数组排序
```py
my_list = [11, 14, 13, 12]

my_list.sort()
# 数组从小到大排序，o(nlogn)
# my_list = [11, 12, 13, 14]

my_list.sort(reverse=True)
# 数组从大到小排序，o(nlogn)
# my_list = [14, 13, 12, 11]
```
